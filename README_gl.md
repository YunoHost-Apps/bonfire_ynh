```bash
sudo yunohost app install https://github.com/YunoHost-Apps/bonfire_ynh/tree/testing --debug
ou
sudo yunohost app upgrade bonfire -u https://github.com/YunoHost-Apps/bonfire_ynh/tree/testing --debug
```
